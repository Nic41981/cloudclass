1./user/register
    1.1 请求参数:
    {
        "name":"XXX",
        "password":"XXX",
        "email":"XXX"
    }
    1.2 正常返回:
    {
        "code":0,
        "msg":"登录成功",
        "date":{
            "name":"XXX",
            "identity":X,
            "email":"XXX"
        }
    }
    1.3 异常返回
        1.3.1 缺少参数
        {
            "code":-3,
            "msg":"缺少参数"
        }
        1.3.2 其他异常
        {
            "code":-1,
            "msg":""
        }
